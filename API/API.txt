swagger: '2.0'
info:
  title: Students WS
  description: A web services for the management of students
  version: "1.0.0"
host: myserver.it
schemes:
  - https
basePath: /
produces:
  - application/json
consumes:
  - application/json
paths:
  /showList:
    get:
      summary: show list of students
      description: it returns the complete lest of all students in the university
      responses:
        200:
          description: the home page
          
  /searchStudent:
    post:
      summary: search a student
      description: it returns the name of a student, given his/her ID
      parameters:
        - name: ID
          in: formData
          description: ID of the student
          required: true
          type: string
      responses:
        200:
          description: student data
          schema: 
              $ref: "#/definitions/student"
        406:
          description: ID of error
          schema: 
            properties:
              error:
                type: integer
        404:
          description: no students with that ID or SSN
          
  /deleteStudent:
    post:
      summary: delete a student
      description: it returns the name of a student, given his/her ID or SSN
      parameters:
        - name: ID
          in: formData
          description: ID of the student
          required: false
          type: string
        - name: SSN
          in: formData
          description: SSN of the student
          required: false
          type: string
      responses:
        200:
          description: name of the student
          schema: 
            $ref: "#/definitions/student"
        406:
          description: ID of error
          schema: 
            properties:
              error:
                type: integer
        404:
          description: no students with that ID or SSN
            
  /searchByMark:
    post:
      summary: return students that match a criteria
      description: it returns a list of students that match a criteria
      parameters:
        - name: criteria
          in: formData
          description: criteria to be matched
          required: true
          type: string
      responses:
        200:
          description: Form with elements
          schema: 
            type: array
            items:
              $ref: "#/definitions/student"
          
definitions:
  student:
    properties:
      ID:
        type: string
        description: ID of the student
      SSN:
        type: string
        description: SSN of the student
      name:
        type: string
        description: name of the student
      address:
        type: string
        description: address of the student
